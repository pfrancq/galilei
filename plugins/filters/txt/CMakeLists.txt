PROJECT(txt)
INCLUDE_DIRECTORIES(.)

SET(SRC txt.cpp)

INCLUDE_DIRECTORIES(${CMAKE_CURRENT_BINARY_DIR})

IF(WIN32)
    ADD_LIBRARY(txt SHARED ${SRC})
    ADD_SOURCE_FILE(txt frontend txt_INCLUDES)
    ADD_FRONTEND(txt SHARED  2 2 2.0.0 lib "r/galilei" txt_INCLUDES)
    SET_TARGET_PROPERTIES(txt PROPERTIES LINK_FLAGS "/NODEFAULTLIB:MSVCRTD;MSVCRT;LIBCMT;LIBCMTD")
    TARGET_LINK_LIBRARIES(txt galilei roptimization rmath rcore)
ELSE(WIN32)
    ADD_LIBRARY(txt SHARED ${SRC})
    ADD_SOURCE_FILE(txt frontend txt_INCLUDES)
    ADD_FRONTEND(txt SHARED  2 2 2.0.0 lib "r/galilei" txt_INCLUDES)
    TARGET_LINK_LIBRARIES(txt galilei roptimization rcore rmath dl)
ENDIF(WIN32)
