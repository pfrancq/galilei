PROJECT(xml)
INCLUDE_DIRECTORIES(.)

SET(SRC gxml.cpp)

INCLUDE_DIRECTORIES(${CMAKE_CURRENT_BINARY_DIR})

IF(WIN32)
    ADD_LIBRARY(xml SHARED ${SRC})
    ADD_SOURCE_FILE(xml frontend xml_INCLUDES)
    ADD_FRONTEND(xml SHARED 2 2 2.0.0 lib "r/galilei" xml_INCLUDES)
    SET_TARGET_PROPERTIES(xml PROPERTIES LINK_FLAGS "/NODEFAULTLIB:MSVCRTD;MSVCRT;LIBCMT;LIBCMTD")
    TARGET_LINK_LIBRARIES(xml galilei roptimization rmath rcore)
ELSE(WIN32)
    ADD_LIBRARY(xml SHARED ${SRC})
    ADD_SOURCE_FILE(xml frontend xml_INCLUDES)
    ADD_FRONTEND(xml SHARED 2 2 2.0.0 lib "r/galilei" xml_INCLUDES)
    TARGET_LINK_LIBRARIES(xml galilei roptimization rcore rmath dl)
ENDIF(WIN32)
