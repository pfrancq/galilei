PROJECT(lang00)
INCLUDE_DIRECTORIES(.)

SET(SRC lang00.cpp)

INCLUDE_DIRECTORIES(${CMAKE_CURRENT_BINARY_DIR})

IF(WIN32)
    ADD_LIBRARY(lang00 STATIC ${SRC})
    SET_TARGET_PROPERTIES(lang00 PROPERTIES LINK_FLAGS "/NODEFAULTLIB:MSVCRTD;MSVCRT;LIBCMT;LIBCMTD")
    TARGET_LINK_LIBRARIES(lang00 galilei roptimization rmath rcore)
ELSE(WIN32)
    ADD_LIBRARY(lang00 SHARED ${SRC})
    ADD_SOURCE_FILE(lang00 frontend lang00_INCLUDES)
    ADD_FRONTEND(lang00 2 2 2.0.0 lib "r/galilei" lang00_INCLUDES)
    TARGET_LINK_LIBRARIES(lang00 galilei roptimization rcore rmath dl)
ENDIF(WIN32)
