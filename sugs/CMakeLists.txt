PROJECT(sugs)
INCLUDE_DIRECTORIES(.)

SET(SRC sugs.cpp)

INCLUDE_DIRECTORIES(${CMAKE_CURRENT_BINARY_DIR})

IF(WIN32)
    ADD_LIBRARY(sugs STATIC ${SRC})
    SET_TARGET_PROPERTIES(sugs PROPERTIES LINK_FLAGS "/NODEFAULTLIB:MSVCRTD;MSVCRT;LIBCMT;LIBCMTD")
    TARGET_LINK_LIBRARIES(sugs galilei roptimization rmath rcore)
ELSE(WIN32)
    ADD_LIBRARY(sugs SHARED ${SRC})
    ADD_SOURCE_FILE(sugs frontend sugs_INCLUDES)
    ADD_FRONTEND(sugs 2 2 2.0.0 lib "r/galilei" sugs_INCLUDES)
    TARGET_LINK_LIBRARIES(sugs galilei roptimization rcore rmath dl)
ENDIF(WIN32)
